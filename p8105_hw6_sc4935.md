Use 5000 bootstrap samples

``` r
boot_straps = 
  weather_df %>% 
  janitor::clean_names() %>%
  modelr::bootstrap(n = 5000) %>% 
  mutate(
    models = map(strap, ~lm(tmax ~ tmin, data = .x) ),
    results = map(models, broom::tidy)) %>% 
  select(-strap, -models) %>% 
  unnest(results)
```
